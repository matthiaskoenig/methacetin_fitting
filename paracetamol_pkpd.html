results_dir = "../results/paracetamol"
apap_dir = '/home/mkoenig/git/acetaminophen/docs/literature/'

def load_apap_data(fid, sep="\t", show=True):
    """ Loads data from given figure/table id.
    Displays the first rows of the dataframe by default.
    """
    df = pd.read_csv(os.path.join(apap_dir, '{}.csv'.format(fid)), sep=sep, comment="#")
    if show == True:
        display(df.head())
        print(fid)
    return df

#--------------------
sid = 'Rawlins1977'
dosing_iv = Dosing(substance="apap", route="iv", dose=1000.0, unit="mg")
tend = 8   # [h]
steps = 300
#--------------------

# data
Rawlins1977_Fig1 = load_apap_data("Rawlins1977_Fig1", show=False)
Rawlins1977_Fig2 = load_apap_data("Rawlins1977_Fig2", show=False)

# simulation (IV 1000[mg])
s_iv = simulate(r, tend=tend, steps=steps, dosing=dosing_iv, parameters=parameters)

oral_doses = []
oral_simulations = []
for dose in Rawlins1977_Fig2.dose.unique():
    dosing = Dosing(substance="apap", route="oral", dose=float(dose), unit="mg")
    oral_doses.append(dosing)
    s = simulate(r, tend=tend, steps=steps, dosing=dosing, parameters=parameters)
    oral_simulations.append(s)

# Figure
fig, (ax1, ax2) = plt.subplots(nrows=1, ncols=2, figsize=(10, 5))
axes = (ax1, ax2)
groups = Rawlins1977_Fig1.dose.unique()

for k, group in enumerate(groups):
    data = Rawlins1977_Fig1[Rawlins1977_Fig1.dose == group]
    add_line('time', "Mve_apap", ax=ax1, s=s_iv, label="apap", color=get_color("apap"))
    ax1.errorbar(data.time, data.apap, data.apap_sd, color=get_color("apap"), 
                 label="apapSD (n=6)", **kwargs_data)
    
    ax1.set_title('{} (1000 [mg] iv)'.format(sid))
    ax1.set_ylabel('Paracetamol [mg/l]')
    ax1.set_ylim(0, 30)

groups = Rawlins1977_Fig2.dose.unique()
for k, group in enumerate(groups):
    data = Rawlins1977_Fig2[Rawlins1977_Fig2.dose == group]
    s = oral_simulations[k]
    add_line('time', "Mve_apap", ax=ax2, s=s, label="apap {}".format(group), color=colors[k])
    ax2.errorbar(data.time, data.apap, data.apap_sd, label="{} apapSD (n=6)".format(group), 
                 color=colors[k], **kwargs_data)
    ax2.set_ylabel('Paracetamol [g/ml]')
    ax2.set_title('{} (500, 1000, 2000 [mg] oral)'.format(sid))
    
for ax in axes:
    ax.set_xlabel("time [h]")
    ax.legend()
    
fig.savefig(os.path.join(results_dir, "apap_{}".format(sid)), dpi=300, bbox_inches="tight")
plt.show()
